
% 题目相关的配置
\newcounter{exgroup}   % 大题的计数器
\newcounter{exitem}[exgroup]   % 小题的计数器
\newcommand{\dati}[1]{ % 大题的标题
  \stepcounter{exgroup}
  \chinese{exgroup}、 #1
}
\newcommand{\xiaoti}{ % 小题的标题
  \stepcounter{exitem}
  \arabic{exitem}、
}

% 添加“选择题”(choice or multiple-choice)选项
\newcommand{\twochoices}[2]{    % 两个选项
  \par\begin{tabular}{*{4}{@{}p{3.4cm}}}
    A. ~#1 & B. ~#2
  \end{tabular}
}
\newcommand{\threechoices}[3]{  % 三个选项
  \par\begin{tabular}{*{4}{@{}p{3.4cm}}}
    A. ~#1 & B. ~#2 & C. ~#3
  \end{tabular}
}
\newcommand{\fourchoices}[4]{   % 四个选项
  \par\begin{tabular}{*{4}{@{}p{3.4cm}}}
    A. ~#1 & B. ~#2 & C. ~#3 & D. ~#4
  \end{tabular}
}
